sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"

select yes on prompt

git clone git@github.com:zsh-users/zsh-autosuggestions.git

edit line with plugins=(git) as

save and exit
